```MySQL

/* Relatorio Whatsapp SEND Exemplo 1 */


SELECT

DATEPART(YEAR,EVENTDATEUTC) as ANO,
DATEPART(MONTH,EVENTDATEUTC) as MES,
COUNT(SENDIDENTIFIER) AS QTDE_TOTAL_WHATSAPP

FROM Tb_whatsapp_report_90d
where convert(varchar,EVENTDATEUTC,3) between '14/11/2022' and '20/11/2022' and TRACKINGTYPE = 'SEND'
GROUP BY 
DATEPART(YEAR,EVENTDATEUTC),
DATEPART(MONTH,EVENTDATEUTC)
```
